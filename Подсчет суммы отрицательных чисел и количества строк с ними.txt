massiv = [[2, 3, -1],
           [-2, 5, 6],
           [0, 0, 0],
           [7, 4, -3]]

otric_count = 0  
sum_count = 0       

# Подсчет строк и суммы отрицательных чисел
for i in massiv:
    if any(num < 0 for num in i): 
        otric_count += 1  
        sum_count += sum(num for num in i if num < 0)  # Суммируем отрицательные числа

print("Количество строк, содержащих хотя бы одно отрицательное число: ", otric_count)
print("Сумма отрицательных чисел в массиве: ", sum_count)
